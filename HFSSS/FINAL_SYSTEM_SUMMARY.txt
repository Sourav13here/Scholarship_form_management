📊 DATABASE SCHEMA
================================================================================

Table: applications
------------------
1.  id                 - INTEGER PRIMARY KEY AUTO INCREMENT
2.  application_id     - TEXT UNIQUE NOT NULL (NCI20240001)
3.  name               - TEXT NOT NULL
4.  class              - TEXT NOT NULL
5.  school             - TEXT NOT NULL
6.  address            - TEXT NOT NULL
7.  contact            - TEXT NOT NULL (10 digits)
8.  alt_contact        - TEXT (optional)
9.  email              - TEXT NOT NULL (unique)
10. achievements       - TEXT (optional)
11. declaration        - INTEGER NOT NULL (1)
12. submission_date    - DATETIME AUTO TIMESTAMP
13. status             - TEXT DEFAULT 'pending'
